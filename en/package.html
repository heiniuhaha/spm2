package.json

spm follows the Common Module Definition packaging standards, compatible with nodejs' package.json.

CMD packaging draft add an additional namespace, which is family. spm add another namespace, which is spm.

A full example of package.json:

{
    "family": "arale",
    "name": "base",
    "version": "1.0.0",
    "description": "base is ....",
    "homepage": "http://aralejs.org/base/",
    "repository": {
        "type": "git",
        "url": "https://github.com/aralejs/base.git"
    },
    "keywords": ["class"],

    "spm": {
        "alias": {
            "class": "arale/class/1.0.0/class",
            "events": "arale/events/1.0.0/events"
        },
        "output": {
            "base.js": "."
        },
        "engines": {
            "seajs": "seajs/seajs/1.2.0/sea.js"
        }
    }
}
